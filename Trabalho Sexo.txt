programa {
    inteiro idade, numeroMoradores
    real salario
    caracter recebeSalario

  funcao inicio() {
    escreva("CENSO 2025 - LEVANTAMENTO DEMOGRÁFICO\n\n")

    escreva("Quantas pessoas vivem na sua residência? ")
    leia(numeroMoradores)

    para(inteiro i = 0; i < numeroMoradores; i++) {
      escreva("\n--- Dados do morador N°", i+1, "---\n")
      coletarInformacoes()
    }
  }

  funcao coletarInformacoes() {
    escreva("Qual a idade do morador? ")
    leia(idade)

    selecionarSexo()

    escreva("O morador recebe salário? (S/N) ")
    leia(recebeSalario)

    se (recebeSalario == 'S' ou recebeSalario == 's') {
      escreva("Informe o valor do salário: ")
      leia(salario)
    }
  }

  funcao selecionarSexo() {
    caracter sexoEscolhido
    escreva("Opções de sexo:\n")
    escreva("H - Homem\n")
    escreva("M - Mulher\n")
    escreva("P - Prefiro não informar\n")

    escreva("Digite a opção correspondente: ")
    leia(sexoEscolhido)
  }
}
